               if(vm.user.id != 1) {
                   filter_resjsontasks = vm.resjsontasks.filter(item => item.id_list == id_list && item.id_user == vm.user.id);

                   filter_resjsontasks = vm.resjsontasks.filter(item => item.id_list == id_list);
